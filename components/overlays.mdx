## DModal

Modal dialog component with animations and composable structure.

### Basic Usage

```tsx
import DModal from '@dynamic-framework/ui-react';
import { AnimatePresence } from 'framer-motion';

function App() {
  const [isOpen, setIsOpen] = useState(false);
  
  return (
    <>
      <DButton text="Open Modal" onClick={() => setIsOpen(true)} />
      
      <AnimatePresence>
        {isOpen && (
          <DModal name="exampleModal" onClose={() => setIsOpen(false)}>
            <DModal.Header>Modal Title</DModal.Header>
            <DModal.Body>
              Modal content goes here
            </DModal.Body>
            <DModal.Footer>
              <DButton text="Close" onClick={() => setIsOpen(false)} />
            </DModal.Footer>
          </DModal>
        )}
      </AnimatePresence>
    </>
  );
}
```

### Props

<ParamField path="name" type="string" required>
  Unique modal identifier
</ParamField>

<ParamField path="staticBackdrop" type="boolean" default="false">
  Prevents closing on backdrop click
</ParamField>

<ParamField path="scrollable" type="boolean" default="false">
  Makes modal body scrollable
</ParamField>

<ParamField path="centered" type="boolean" default="false">
  Vertically centers modal
</ParamField>

<ParamField path="fullScreen" type="boolean" default="false">
  Full screen modal
</ParamField>

<ParamField path="fullScreenFrom" type="ModalFullScreenFrom">
  Breakpoint for full screen - 'sm', 'md', 'lg', 'xl', or 'xxl'
</ParamField>

<ParamField path="size" type="ModalSize">
  Modal size - 'sm', 'lg', or 'xl'
</ParamField>

<ParamField path="transition" type="Transition">
  Custom Framer Motion transition
</ParamField>

### Subcomponents

#### DModal.Header
Modal header section

#### DModal.Body
Modal body section (main content)

#### DModal.Footer
Modal footer section (typically for actions)

## DOffcanvas

Offcanvas panel that slides in from any edge of the viewport.

### Basic Usage

```tsx
import DOffcanvas from '@dynamic-framework/ui-react';
import { AnimatePresence } from 'framer-motion';

function App() {
  const [isOpen, setIsOpen] = useState(false);
  
  return (
    <>
      <DButton text="Open Menu" onClick={() => setIsOpen(true)} />
      
      <AnimatePresence>
        {isOpen && (
          <DOffcanvas name="menu" openFrom="end">
            <DOffcanvas.Header>
              <h5>Menu</h5>
              <DButtonIcon 
                icon="X" 
                ariaLabel="Close"
                onClick={() => setIsOpen(false)} 
              />
            </DOffcanvas.Header>
            <DOffcanvas.Body>
              Navigation content
            </DOffcanvas.Body>
          </DOffcanvas>
        )}
      </AnimatePresence>
    </>
  );
}
```

### Props

<ParamField path="name" type="string" required>
  Unique offcanvas identifier
</ParamField>

<ParamField path="openFrom" type="OffcanvasPositionToggleFrom" default="end">
  Edge to open from - 'top', 'bottom', 'start', or 'end'
</ParamField>

<ParamField path="staticBackdrop" type="boolean" default="false">
  Prevents closing on backdrop click
</ParamField>

<ParamField path="scrollable" type="boolean" default="false">
  Allows body scroll when offcanvas is open
</ParamField>

<ParamField path="transition" type="Transition">
  Custom Framer Motion transition
</ParamField>

### Subcomponents

#### DOffcanvas.Header
Offcanvas header section

#### DOffcanvas.Body
Offcanvas body section

#### DOffcanvas.Footer
Offcanvas footer section

## DPopover

Popover component with floating UI positioning.

### Basic Usage

```tsx
import DPopover from '@dynamic-framework/ui-react';

function App() {
  const [open, setOpen] = useState(false);
  
  return (
    <DPopover
      open={open}
      setOpen={setOpen}
      renderComponent={(isOpen) => (
        <DButton text="Toggle Popover" />
      )}
    >
      <div className="p-3">
        <h6>Popover Title</h6>
        <p>Popover content goes here</p>
      </div>
    </DPopover>
  );
}
```

### Props

<ParamField path="open" type="boolean" required>
  Controls popover visibility
</ParamField>

<ParamField path="setOpen" type="(open: boolean) => void">
  Callback to update open state
</ParamField>

<ParamField path="renderComponent" type="(open: boolean) => ReactElement" required>
  Function that renders the trigger element
</ParamField>

<ParamField path="adjustContentToRender" type="boolean" default="false">
  Adjusts popover width to content
</ParamField>

### Examples

<Tabs>
  <Tab title="Basic">
    ```tsx
    function BasicPopover() {
      const [open, setOpen] = useState(false);
      
      return (
        <DPopover
          open={open}
          setOpen={setOpen}
          renderComponent={(isOpen) => (
            <DButtonIcon 
              icon="Info" 
              ariaLabel="Info"
            />
          )}
        >
          <div className="p-3">
            <p>Additional information here</p>
          </div>
        </DPopover>
      );
    }
    ```
  </Tab>

    ```tsx
        <DButton text="Profile" variant="link" />
    ```

```typescript
type Props = BaseProps & PropsWithChildren<{
  renderComponent: (open: boolean) => ReactElement<unknown>;
  open: boolean;
  setOpen?: (open: boolean) => void;
  adjustContentToRender?: boolean;
}>;
```

## DTooltip

Tooltip component with customizable triggers and positioning.

### Basic Usage

```tsx
import DTooltip from '@dynamic-framework/ui-react';

function App() {
  return (
    <DTooltip 
      Component={<DButton text="Hover me" />}
      placement="top"
    >
      This is a tooltip
    </DTooltip>
  );
}
```

### Props

<ParamField path="Component" type="ReactNode" required>
  Element that triggers the tooltip
</ParamField>

<ParamField path="placement" type="Placement" default="top">
  Tooltip position - 'top', 'bottom', 'left', 'right', etc.
</ParamField>

<ParamField path="withHover" type="boolean" default="true">
  Trigger on hover
</ParamField>

<ParamField path="withFocus" type="boolean" default="false">
  Trigger on focus
</ParamField>

<ParamField path="withClick" type="boolean" default="false">
  Trigger on click
</ParamField>

<ParamField path="open" type="boolean" default="false">
  Controlled open state
</ParamField>

<ParamField path="size" type="ComponentSize">
  Tooltip size - 'sm' or 'lg'
</ParamField>

<ParamField path="offSet" type="number">
  Distance from trigger element
</ParamField>

<ParamField path="padding" type="number">
  Padding for shift middleware
</ParamField>

### Examples

<Tabs>
  <Tab title="Basic">

### TypeScript Interface

```typescript
export type Props = BaseProps & PropsWithChildren<{
  childrenClassName?: string;
  offSet?: number;
  padding?: number;

  withFocus?: boolean;
  withClick?: boolean;
  open?: boolean;
  size?: ComponentSize;
  placement?: Placement;
  Component: ReactNode;
}>;
```

## DDropdown

Dropdown menu component with action items.

### Basic Usage

```tsx
import DDropdown from '@dynamic-framework/ui-react';

function App() {
  const actions = [
    {
      label: 'Edit',
      icon: 'Edit',
      onClick: () => console.log('Edit clicked')
    },
    {
      label: 'Delete',
      icon: 'Trash',
      color: 'danger' as const,
      onClick: () => console.log('Delete clicked')
    },
  ];
  
  return <DDropdown actions={actions} />;
}
```

### Props

<ParamField path="actions" type="DropdownAction[]" required>
  Array of dropdown actions
</ParamField>

<ParamField path="dropdownToggle" type="(props: { open: boolean, toggle: () => void }) => ReactNode">
  Custom toggle element renderer
</ParamField>

<ParamField path="className" type="string">
  Additional CSS classes
</ParamField>

### DropdownAction Interface

<ParamField path="label" type="string" required>
  Action label
</ParamField>

<ParamField path="onClick" type="() => void">
  Click handler for button actions
</ParamField>

<ParamField path="href" type="string">
  Link URL (renders as anchor)
</ParamField>

<ParamField path="icon" type="string">
  Icon name
</ParamField>

<ParamField path="color" type="'default' | 'danger' | 'success' | 'warning' | 'info'">
  Action color variant
</ParamField>

<ParamField path="disabled" type="boolean">
  Disables the action
</ParamField>

<ParamField path="isDivider" type="boolean">
  Renders as divider instead of action
</ParamField>

### Examples

### TypeScript Interface

```typescript
export type DropdownAction = {

  href?: string;
```
