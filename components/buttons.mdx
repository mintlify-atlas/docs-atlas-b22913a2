## DButton

Primary button component with support for variants, colors, sizes, icons, and loading states. Can render as a button or anchor element.

### Basic Usage

```tsx
import DButton from '@dynamic-ui/components';

function App() {
  return (
    <DButton 
      text="Click me"
      color="primary"
      onClick={() => console.log('Clicked!')}
    />
  );
}
```

### Props

<ParamField path="text" type="string">
  Button text content
</ParamField>

<ParamField path="color" type="ComponentColor" default="primary">
  Button color theme (primary, secondary, success, danger, warning, info, etc.)
</ParamField>

<ParamField path="size" type="ComponentSize">
  Button size - 'sm' or 'lg'
</ParamField>

<ParamField path="variant" type="ButtonVariant">
  Button variant - 'outline' or 'link'
</ParamField>

<ParamField path="loading" type="boolean" default="false">
  Shows loading spinner when true
</ParamField>

<ParamField path="loadingText" type="string">
  Text to display during loading state
</ParamField>

<ParamField path="loadingAriaLabel" type="string">
  Aria label for loading state
</ParamField>

<ParamField path="disabled" type="boolean" default="false">
  Disables the button
</ParamField>

<ParamField path="iconStart" type="string">
  Icon name to display at start of button
</ParamField>

<ParamField path="iconEnd" type="string">
  Icon name to display at end of button
</ParamField>

<ParamField path="href" type="string">
  If provided, renders as an anchor tag instead of button
</ParamField>

<ParamField path="target" type="string">
  Link target attribute (when used with href)
</ParamField>

<ParamField path="type" type="ButtonType" default="button">
  Button type - 'button', 'submit', or 'reset'
</ParamField>

```tsx
<DButton 
  text="Submit" 
  loading={isSubmitting}
  loadingText="Submitting..."
  color="primary" 
/>
```

### As Link

```tsx

  text="Visit Page" 
  href="/page" 
  target="_blank"
  rel="noopener noreferrer"
  color="primary" 
/>
```

## DButtonIcon

Icon-only button component for compact interfaces. Supports all button states and can render as button or anchor.

### Basic Usage

```tsx
import DButtonIcon from '@dynamic-ui/components';

function App() {
  return (
    <DButtonIcon 
      icon="Settings"
      ariaLabel="Open settings"
      color="primary"
      onClick={() => console.log('Settings clicked')}
    />
  );
}
```

### Props

<ParamField path="icon" type="string" required>
  Icon name to display
</ParamField>

<ParamField path="ariaLabel" type="string">
  Accessible label for screen readers (highly recommended)
</ParamField>

<ParamField path="color" type="ComponentColor" default="primary">
  Button color theme
</ParamField>

<ParamField path="size" type="ComponentSize">
  Button size - 'sm' or 'lg'
</ParamField>

<ParamField path="variant" type="ButtonVariant">
  Button variant - 'outline' or 'link'
</ParamField>

<ParamField path="loading" type="boolean" default="false">
  Shows loading spinner when true
</ParamField>

<ParamField path="disabled" type="boolean" default="false">
  Disables the button
</ParamField>

<ParamField path="state" type="InputState">
  Visual state - 'focus-visible', 'hover', 'active', or 'disabled'
</ParamField>

<ParamField path="stopPropagationEnabled" type="boolean" default="true">
  Whether to stop event propagation on click
</ParamField>

<ParamField path="href" type="string">
  If provided, renders as anchor tag
</ParamField>
