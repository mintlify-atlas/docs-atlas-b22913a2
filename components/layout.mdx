## DCard

Card container component with composable header, body, and footer sections.

### Basic Usage

```tsx
import DCard from '@dynamic-ui/components';

function App() {
  return (
    <DCard>
      <DCard.Header>Card Header</DCard.Header>
      <DCard.Body>Card content goes here</DCard.Body>
      <DCard.Footer>Card Footer</DCard.Footer>
    </DCard>
  );
}
```

### Props

<ParamField path="className" type="string">
  Additional CSS classes
</ParamField>

<ParamField path="style" type="CSSProperties">
  Inline styles
</ParamField>

<ParamField path="dataAttributes" type="DataAttributes">
  Data attributes
</ParamField>

### Subcomponents

#### DCard.Header
Card header section

#### DCard.Body
Card body section (main content area)

#### DCard.Footer
Card footer section

## DLayout

Grid-based layout system with responsive gap control.

### Basic Usage

```tsx
import DLayout from '@dynamic-ui/components';

function App() {
  return (
    <DLayout gap={3} columns={12}>
      <DLayout.Pane columnSpan={6}>
        Left Column
      </DLayout.Pane>
      <DLayout.Pane columnSpan={6}>
        Right Column
      </DLayout.Pane>
    </DLayout>
  );
}
```

### Props

<ParamField path="gap" type="number">
  Gap spacing on scale from 0 to 30
</ParamField>

<ParamField path="gapSm" type="number">
  Gap spacing for small breakpoint
</ParamField>

<ParamField path="gapMd" type="number">
  Gap spacing for medium breakpoint
</ParamField>

<ParamField path="gapLg" type="number">
  Gap spacing for large breakpoint
</ParamField>

<ParamField path="gapXl" type="number">
  Gap spacing for extra large breakpoint
</ParamField>

<ParamField path="gapXxl" type="number">
  Gap spacing for 2x extra large breakpoint
</ParamField>

<ParamField path="columns" type="number">
  Number of columns in grid (sets --bs-columns CSS variable)
</ParamField>

### DLayout.Pane Props

The `DLayout.Pane` component accepts responsive column span props:

<ParamField path="columnSpan" type="number">
  Default column span
</ParamField>

<ParamField path="columnSpanSm" type="number">
  Column span for small breakpoint
</ParamField>

<ParamField path="columnSpanMd" type="number">
  Column span for medium breakpoint
</ParamField>

<ParamField path="columnSpanLg" type="number">
  Column span for large breakpoint
</ParamField>

<ParamField path="columnSpanXl" type="number">
  Column span for extra large breakpoint
</ParamField>

<ParamField path="columnSpanXxl" type="number">
  Column span for 2x extra large breakpoint
</ParamField>

## DBox

Simple container component for wrapping content.

### Basic Usage

```tsx
import DBox from '@dynamic-ui/components';

function App() {
  return (
    <DBox className="p-4 bg-light rounded">
      <h3>Content Title</h3>
      <p>Content goes here</p>
    </DBox>
  );
}
```

### Props

<ParamField path="className" type="string">
  Additional CSS classes
</ParamField>

<ParamField path="style" type="CSSProperties">
  Inline styles
</ParamField>

<ParamField path="dataAttributes" type="DataAttributes">
  Data attributes
</ParamField>

### Examples

```tsx
// Simple container
<DBox className="p-3">
  <p>Simple box with padding</p>
</DBox>

// Custom styled box
<DBox 
  className="border rounded shadow-sm" 
  style={{ backgroundColor: '#f8f9fa' }}
>
  <h4>Custom Box</h4>
  <p>With custom styling</p>
</DBox>

// Data attributes
<DBox 
  dataAttributes={{
    'data-testid': 'content-box',
    'data-section': 'main'
  }}
>
  <p>Box with data attributes</p>
</DBox>
```

### TypeScript Interface

```typescript
type Props = PropsWithChildren<BaseProps>;

export default function DBox(props: Props): JSX.Element;
```
